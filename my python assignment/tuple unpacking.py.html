# tuple unpacking
tup_nbr = (4, 5, 8, 9)
for item in tup_nbr:
    print(item)
list_tpl = [(4, 5), (8, 9)]
for item in list_tpl:
    print(item)

list1 = [(2, 3, 4), (9, 8, 7), (1, 5, 6)]
for item in list1:
    print(item)
else:
    print("list printed")

for a, b, c in list1:
    print(a)

# for loop for dictionary
dict_1 = {'bts': 2013, 'got7': 2014, 'bigbang': 2006}
print(dict_1)
for item in dict_1.items():
    print(item)
else:print("dict unpacked")
for x,y in dict_1.items():
    print(x)
    print(y)
